# Combinatorica Game Settings
#
# Combinatorica bundles ~80 mods organized by category. Each mod is a
# "domain" that can be individually enabled/disabled via its
# combinatorica_<name>_enabled setting below. Defaults are all true — the
# game works out of the box; tailor for your server.
#
# Some coarse feature flags (combinatorica_hunger_enabled,
# combinatorica_mobs_enabled, combinatorica_techage_enabled,
# combinatorica_nether_enabled, combinatorica_extra_biomes) disable entire
# subsystems at once and take precedence over individual per-mod flags.

[Gameplay]

# Enable the hunger system
combinatorica_hunger_enabled (Enable hunger) bool true

# Enable mob spawning (disables the entire mobs_pack)
combinatorica_mobs_enabled (Enable mobs) bool true

# Enable PvP combat
enable_pvp (Enable PvP) bool true

[Starting Items]

# Give new players starter items
give_initial_stuff (Give initial stuff) bool true

# Comma-separated list of items to give new players
# Format: itemname count, itemname count, ...
initial_stuff (Initial items) string default:pick_wood 1,default:torch 8,default:apple 4

[Technology]

# Enable the Techage technology progression (disables the entire techage_pack)
combinatorica_techage_enabled (Enable Techage) bool true

[World Generation]

# Enable the Nether dimension
combinatorica_nether_enabled (Enable Nether) bool true

# Enable extra biomes beyond the base set (disables the entire worldgen_pack)
combinatorica_extra_biomes (Enable extra biomes) bool true
